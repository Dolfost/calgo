116template<typename T>
117void mat<T>::resize(const size_type& rows, const size_type& cols, bool copy) {
118 if (rows == this->m_rows and cols == this->m_cols)
119 return;
120
121 size_type size = rows*cols;
122 if (copy) {
123 value_type* newMat = new value_type[size];
124 size_type stopRow = std::min(rows, this->m_rows);
125 size_type stopCol = std::min(cols, this->m_cols);
126 for (size_type i = 0; i < stopRow; i++)
127 for (size_type j = 0; j < stopCol; j++)
128 *CALGO_MAT_INDEX(newMat, cols, 1, i, j) =
129 this->el(i, j);
130 for (size_type i = 0; i < rows - stopRow; i++)
131 for (size_type j = 0; j < cols - stopCol; j++)
132 *CALGO_MAT_INDEX(newMat, cols, 1, i, j) = 0;
133 delete[] this->m_mat;
134 this->m_mat = newMat;
135 } else {
136 delete[] this->m_mat;
137 this->m_mat = new value_type[size];
138 }
139
140 this->m_rows = rows;
141 this->m_cols = cols;
142 this->m_dist = 1;
143}
